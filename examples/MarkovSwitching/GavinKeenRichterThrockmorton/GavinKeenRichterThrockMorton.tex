\end{equation}

\end{equation}

\begin{equation}
\beta _{t}=\beta

\begin{equation}
r_{t}^{k}=\frac{1}{\beta _{t}}-1+\delta 
\end{equation}

\[

\]

\begin{equation}
r_{t}=\max \left\{ 1,\bar{r}\right\}

\begin{equation}

\end{equation}

\begin{equation}
w_{t}=\left[ \frac{\Psi _{t}z_{t}}{\left( r_{t}^{k}\right) ^{\alpha }\left(
